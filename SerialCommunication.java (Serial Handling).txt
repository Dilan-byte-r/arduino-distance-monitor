import com.fazecast.jSerialComm.*;
import javax.swing.*;

/**
 * Handles serial communication with Arduino
 * Uses jSerialComm library for cross-platform compatibility
 * 
 * @author Dilan Janaka Hewarahinduwage
 * @version 1.0
 */
public class SerialCommunication {
    
    private SerialPort serialPort;
    private Thread readingThread;
    private volatile boolean running = false;
    private DistanceMonitorGUI gui;
    
    // Buffer for incoming data
    private StringBuilder buffer = new StringBuilder();
    
    public SerialCommunication(DistanceMonitorGUI gui) {
        this.gui = gui;
    }
    
    /**
     * Gets list of available COM ports
     */
    public String[] getAvailablePorts() {
        SerialPort[] ports = SerialPort.getCommPorts();
        String[] portNames = new String[ports.length];
        for (int i = 0; i < ports.length; i++) {
            portNames[i] = ports[i].getSystemPortName();
        }
        return portNames;
    }
    
    /**
     * Connects to the specified COM port
     */
    public boolean connect(String portName, int baudRate) throws Exception {
        // Find the port
        SerialPort[] ports = SerialPort.getCommPorts();
        SerialPort foundPort = null;
        
        for (SerialPort port : ports) {
            if (port.getSystemPortName().equals(portName)) {
                foundPort = port;
                break;
            }
        }
        
        if (foundPort == null) {
            throw new Exception("Port not found: " + portName);
        }
        
        serialPort = foundPort;
        serialPort.setBaudRate(baudRate);
        serialPort.setNumDataBits(8);
        serialPort.setNumStopBits(1);
        serialPort.setParity(SerialPort.NO_PARITY);
        serialPort.setFlowControl(SerialPort.FLOW_CONTROL_DISABLED);
        
        // Open the port
        if (!serialPort.openPort()) {
            throw new Exception("Failed to open port: " + portName);
        }
        
        // Start reading thread
        running = true;
        readingThread = new Thread(this::readData);
        readingThread.start();
        
        return true;
    }
    
    /**
     * Disconnects from the serial port
     */
    public void disconnect() {
        running = false;
        
        if (readingThread != null) {
            try {
                readingThread.interrupt();
                readingThread.join(1000);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
            readingThread = null;
        }
        
        if (serialPort != null && serialPort.isOpen()) {
            serialPort.closePort();
            serialPort = null;
        }
    }
    
    /**
     * Sends a command to Arduino
     */
    public void sendCommand(String command) {
        if (serialPort == null || !serialPort.isOpen()) {
            gui.logEvent("Error: Not connected");
            return;
        }
        
        try {
            byte[] data = command.getBytes();
            serialPort.writeBytes(data, data.length);
            gui.logEvent("Sent: " + command.trim());
        } catch (Exception e) {
            gui.logEvent("Send error: " + e.getMessage());
        }
    }
    
    /**
     * Reads data from Arduino (runs in separate thread)
     */
    private void readData() {
        byte[] readBuffer = new byte[1024];
        
        while (running) {
            try {
                if (serialPort != null && serialPort.isOpen()) {
                    int numBytes = serialPort.readBytes(readBuffer, readBuffer.length);
                    
                    if (numBytes > 0) {
                        String data = new String(readBuffer, 0, numBytes);
                        processIncomingData(data);
                    }
                }
                
                Thread.sleep(10);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                break;
            } catch (Exception e) {
                if (running) {
                    gui.logEvent("Read error: " + e.getMessage());
                }
                break;
            }
        }
    }
    
    /**
     * Processes incoming data, handling line-based protocol
     */
    private void processIncomingData(String data) {
        // Append to buffer
        buffer.append(data);
        
        // Process complete lines
        String currentLine;
        while ((currentLine = getNextLine()) != null) {
            gui.handleSerialData(currentLine.trim());
        }
    }
    
    /**
     * Extracts the next complete line from the buffer
     */
    private String getNextLine() {
        int index = buffer.indexOf("\n");
        if (index >= 0) {
            String line = buffer.substring(0, index);
            buffer.delete(0, index + 1);
            return line;
        }
        return null;
    }
    
    /**
     * Checks if connected
     */
    public boolean isConnected() {
        return serialPort != null && serialPort.isOpen();
    }
}
